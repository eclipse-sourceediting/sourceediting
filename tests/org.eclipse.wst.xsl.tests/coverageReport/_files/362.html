380                        SemanticContent other = (SemanticContent) arg0;
381                        /* Equal weighting for the same types */
382                        if (this.type.equals(other.type))
383                                return 0;
384                        /* Subtypes have more weight than base types */
385                        if (this.type.isKindOf(other.type))
386                                return 1;
387                        return -1;
388                }

400        private HighlightingStyle createHighlightingStyle(ISemanticHighlighting highlighting) {
401                IPreferenceStore store = highlighting.getPreferenceStore();
402                HighlightingStyle highlightingStyle = null;
403                if (store != null) {
404                        int style = getBoolean(store, highlighting.getBoldPreferenceKey()) ? SWT.BOLD : SWT.NORMAL;
405                        
406                        if (getBoolean(store, highlighting.getItalicPreferenceKey()))
407                                style |= SWT.ITALIC;
408                        if (getBoolean(store, highlighting.getStrikethroughPreferenceKey()))
409                                style |= TextAttribute.STRIKETHROUGH;
410                        if (getBoolean(store, highlighting.getUnderlinePreferenceKey()))
411                                style |= TextAttribute.UNDERLINE;
412                        store.addPropertyChangeListener(fHighlightingChangeListener);
413                        boolean isEnabled = getBoolean(store, highlighting.getEnabledPreferenceKey());
414                        String rgbString = getString(store, highlighting.getColorPreferenceKey());
415                        Color color = null;
416                        
417                        if (rgbString != null)
418                                color = EditorUtility.getColor(ColorHelper.toRGB(rgbString));
419                        
420                        highlightingStyle = new HighlightingStyle(new TextAttribute(color, null, style), isEnabled);
421                }
422                return highlightingStyle;
423        }

497        /**
498         * Handles property change events for individual semantic highlightings.
499         * @param event
500         */
501        private void handleHighlightingPropertyChange(PropertyChangeEvent event) {
502                String property = event.getProperty();
503                if (property == null)
504                        return;
505                
506                boolean refreshRequired = false;
507                
508                for (int i = 0; i < fHighlightings.length; i++) {
509                        ISemanticHighlighting highlighting = fHighlightings[i];
510                        
511                        if (property.equals(highlighting.getBoldPreferenceKey())) {
512                                adaptToTextStyleChange(fHighlightingStyles[i], event, SWT.BOLD);
513                                fPresenter.highlightingStyleChanged(fHighlightingStyles[i]);
514                                refreshRequired = true;
515                                continue;
516                        }
517                        
518                        if (property.equals(highlighting.getColorPreferenceKey())) {
519                                adaptToTextForegroundChange(fHighlightingStyles[i], event);
520                                fPresenter.highlightingStyleChanged(fHighlightingStyles[i]);
521                                refreshRequired = true;
522                                continue;
523                        }
524                        
525                        if (property.equals(highlighting.getEnabledPreferenceKey())) {
526                                adaptToEnablementChange(fHighlightingStyles[i], event);
527                                fPresenter.highlightingStyleChanged(fHighlightingStyles[i]);
528                                refreshRequired = true;
529                                continue;
530                        }
531                        
532                        if (property.equals(highlighting.getItalicPreferenceKey())) {
533                                adaptToTextStyleChange(fHighlightingStyles[i], event, SWT.ITALIC);
534                                fPresenter.highlightingStyleChanged(fHighlightingStyles[i]);
535                                refreshRequired = true;
536                                continue;
537                        }
538                        
539                        if (property.equals(highlighting.getStrikethroughPreferenceKey())) {
540                                adaptToTextStyleChange(fHighlightingStyles[i], event, TextAttribute.STRIKETHROUGH);
541                                fPresenter.highlightingStyleChanged(fHighlightingStyles[i]);
542                                refreshRequired = true;
543                                continue;
544                        }
545                        
546                        if (property.equals(highlighting.getUnderlinePreferenceKey())) {
547                                adaptToTextStyleChange(fHighlightingStyles[i], event, TextAttribute.UNDERLINE);
548                                fPresenter.highlightingStyleChanged(fHighlightingStyles[i]);
549                                refreshRequired = true;
550                                continue;
551                        }
552                }
553                
554                if (refreshRequired && fReconciler != null)
555                        fReconciler.refresh();
556        }

592        private void adaptToTextForegroundChange(HighlightingStyle highlighting, PropertyChangeEvent event) {
593                RGB rgb = null;
594 
595                Object value = event.getNewValue();
596                if (value instanceof RGB)
597                        rgb= (RGB) value;
598                else if (value instanceof String)
599                        rgb= ColorHelper.toRGB( (String) value);
600 
601                if (rgb != null) {
602                        Color color= EditorUtility.getColor(rgb);
603                        TextAttribute oldAttr= highlighting.getTextAttribute();
604                        highlighting.setTextAttribute(new TextAttribute(color, oldAttr.getBackground(), oldAttr.getStyle()));
605                }

608        private void adaptToTextStyleChange(HighlightingStyle highlighting, PropertyChangeEvent event, int styleAttribute) {
609                boolean eventValue = false;
610                Object value = event.getNewValue();
611                if (value instanceof Boolean)
612                        eventValue = ((Boolean) value).booleanValue();
613                else if (IPreferenceStore.TRUE.equals(value))
614                        eventValue = true;
615 
616                TextAttribute oldAttr = highlighting.getTextAttribute();
617                boolean activeValue = (oldAttr.getStyle() & styleAttribute) == styleAttribute;
618 
619                if (activeValue != eventValue)
620                        highlighting.setTextAttribute(new TextAttribute(oldAttr.getForeground(), oldAttr.getBackground(), eventValue ? oldAttr.getStyle() | styleAttribute : oldAttr.getStyle() & ~styleAttribute));
621        }
